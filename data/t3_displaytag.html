/** * This method implements the onClick method of DisplayTag anchor tags so the table is * refreshed via an ajax request rather than a simple GET request. */ function displayTagOnClick(tblId, a) { if (typeof(a) != 'string' && $(a).hasClass('disabled')) return false; var url = (typeof(a) == 'string') ? a : a.href; $.get(url, function(html){ var newTbl = $($.parseHTML(html)).find("#" + tblId); // replace table with new table returned by controller $("#" + tblId).html(newTbl.html()); $("#" + tblId).next().html(newTbl.next().html()); }); return false; }
